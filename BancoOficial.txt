drop database MyTeach;
create database MyTeach;
use Myteach;

create table pagamento_prof 
(
id_pagtprof int auto_increment not null,
tipo_pagtprof text not null,
banco_pagtprof text not null,
agencia_pagtprof text not null,
conta_pagtprof text not null,
pagt_pagtprof  text not null,
data_pagtprof text not null,

primary key (id_pagtprof)
);

create table professor
(
id_prof int auto_increment not null,
nome_prof text not null,
email_prof text not null,
dataNasc_prof text not null,
cpf_prof text not null,
sexo_prof text not null,
cep_prof text not null,
cidade_prof text not null,
rua_prof text not null,
bairro_prof text not null,
num_prof text not null,
tel_prof text not null,
pass_prof text not null,
imagem_prof text not null,
diplomafrente_prof text not null,
diplomatras_prof text not null,
id_pagtprof int not null,
status_prof text not null,

primary key (id_prof),
foreign key(id_pagtprof) references pagamento_prof(id_pagtprof)
);





create table aluno
(
id_user int auto_increment not null,

nome_user text not null,
email_user text not null,
dataNasc_user text not null,
cpf_user text not null,
sexo_user text not null,
cep_user text not null,
cidade_user text not null,
rua_user text not null,
bairro_user text not null,
num_user text not null,
tel_user text not null,
pass_user text not null,
imagem_user text not null,
status_user text not null,

primary key (id_user)

);


create table adm
(
id_adm int auto_increment not null,
nome_adm text not null,
email_adm text not null,
dataNasc_adm text not null,
cpf_adm text not null,
sexo_adm text not null,
cep_adm text not null,
cidade_adm text not null,
rua_adm text not null,
bairro_adm text not null,
num_adm text not null,
tel_adm text not null,
imagem_adm text not null,
status_adm text not null,
pergsecre_adm text not null,
datacad_adm text not null,
acesso_adm text not null,
pass_adm text not null,
pass1_adm text,
pass2_adm text,


primary key (id_adm)
);

create table categoria
(
id_cat int auto_increment not null,
nome_cat text not null,
status_cat text not null,

primary key (id_cat)

);

create table curso
(
id_curso int auto_increment not null,
id_cat int not null,
id_prof int not null,
nome_curso text not null,
valor_curso text not null,
desc_curso text not null,
tam_curso text not null,
imagem_curso text not null,
contem_curso text not null,
arquivo_curso text not null,
status_curso text not null,



primary key (id_curso),
foreign key(id_cat) references categoria(id_cat),
foreign key(id_prof) references professor(id_prof)

);

create table compra
(
id_compra int auto_increment not null,
id_user int not null,
id_curso int not null,
valor_compra text not null,
data_compra text not null,

primary key (id_compra),

foreign key(id_user) references aluno(id_user),
foreign key(id_curso) references curso(id_curso)
);



create table carteira
(
id_cart int auto_increment not null,
id_user int not null,
valor_cart text not null,
qnt_cart text not null,

primary key(id_cart),
foreign key(id_user) references aluno(id_user)
);


create table auditoriaA
(
    id_audA int auto_increment not null,
    datahora_audA text not null,
    id_adm int not null,
    acao_audA text not null, 
    desc_audA text not null,
 
    foreign key (id_adm)references adm(id_adm),
    primary key(id_audA)

create table auditoriaP
(
    id_audP int auto_increment not null,
    datahora_audP text not null,
    id_prof int not null,
    acao_audP text not null, 
    desc_audP text not null, 

    foreign key (id_prof)references professor(id_prof),
    primary key(id_audP)
);

create table auditoriaU
(
    id_audU int auto_increment not null,
    datahora_audU text not null,
    id_user int not null,
    acao_audU text not null, 
    desc_audU text not null, 

    foreign key (id_user)references aluno(id_user),
    primary key(id_audU)
);
